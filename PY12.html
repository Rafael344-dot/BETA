    if idade >= 18:
    print("Você é maior de idade.")
elif idade == 16:
    print("Você tem 16 anos!")
else:
    print("Você é menor de idade.")

    print("Você é maior de idade.")
if idade < 18:
    print("Você é menor de idade.")

    print("Você tem 18 anos.")
  • D) 
    if idade == 16:
    print("Você tem 16 anos.")
elif idade == 17:
